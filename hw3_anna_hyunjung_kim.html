percent_cols <- c(

remove_special <- function(x) {
  x <- gsub("%", "", x)                         
  x <- gsub("\\*", "", x)                       
  x <- str_extract(x, "-?\\d+(,\\d{3})*")       
  as.numeric(x)
}


for (col in percent_cols) {
  hw3[[col]] <- remove_special(hw3[[col]])
}
